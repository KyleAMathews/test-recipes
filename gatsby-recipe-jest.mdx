Adding some jest test files for you to play with

  return a + b;
}

</File>

<File path="src/jest-example/sum.test.js">
const sum = require('./sum');

test('adds 1 + 2 to equal 3', () => {
  expect(sum(1, 2)).toBe(3);
});
